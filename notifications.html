============================================================ 【2】通知作成用コード いいね・コメント・フォロー通知を発生させるための追加コードです。 これは notifications.html ではなく、既存の index.html / detail.html / profile.html などの該当処理に組み込んでください。 // 通知作成用コード // これは各ページの既存処理に追加するコードです。 // notifications.html とは別に、いいね・コメント・フォローの処理があるページへ組み込んでください。 import { collection, doc, getDoc, addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/12.13.0/firebase-firestore.js"; function getUserDisplayNameFromData(data){ if(!data){ return ""; } return ( data.displayName || data.nickname || data.nickName || data.userName || data.name || "" ); } async function getCurrentUserName(db, currentUser){ if(!currentUser){ return "匿名ユーザー"; } try{ const meSnap = await getDoc(doc(db, "users", currentUser.uid)); if(meSnap.exists()){ return getUserDisplayNameFromData(meSnap.data()) || "匿名ユーザー"; } }catch(error){ console.warn("自分のユーザー名取得に失敗しました", error); } return currentUser.displayName || "匿名ユーザー"; } async function createNotification(db, { toUserId, fromUserId, fromUserName, type, albumId = "", albumTitle = "", commentText = "" }){ if(!toUserId || !fromUserId){ return; } // 自分自身への通知は作らない if(toUserId === fromUserId){ return; } await addDoc(collection(db, "notifications"), { toUserId, fromUserId, fromUserName, type, albumId, albumTitle, commentText, read:false, createdAt:serverTimestamp() }); } /* ============================================================ 1. いいね通知 toggleLike(albumId) の「いいね追加」側で setDoc 後に入れる ============================================================ */ async function notifyLike(db, currentUser, albumId){ if(!currentUser || !albumId){ return; } const albumSnap = await getDoc(doc(db, "albums", albumId)); if(!albumSnap.exists()){ return; } const albumData = albumSnap.data(); const fromUserName = await getCurrentUserName(db, currentUser); await createNotification(db, { toUserId: albumData.userId, fromUserId: currentUser.uid, fromUserName, type: "like", albumId, albumTitle: albumData.title || "" }); } /* ============================================================ 2. コメント通知 コメントを addDoc した直後に入れる ============================================================ */ async function notifyComment(db, currentUser, albumId, commentText){ if(!currentUser || !albumId){ return; } const albumSnap = await getDoc(doc(db, "albums", albumId)); if(!albumSnap.exists()){ return; } const albumData = albumSnap.data(); const fromUserName = await getCurrentUserName(db, currentUser); await createNotification(db, { toUserId: albumData.userId, fromUserId: currentUser.uid, fromUserName, type: "comment", albumId, albumTitle: albumData.title || "", commentText: commentText || "" }); } /* ============================================================ 3. フォロー通知 follows に setDoc した直後に入れる ============================================================ */ async function notifyFollow(db, currentUser, targetUserId){ if(!currentUser || !targetUserId){ return; } const fromUserName = await getCurrentUserName(db, currentUser); await createNotification(db, { toUserId: targetUserId, fromUserId: currentUser.uid, fromUserName, type: "follow" }); }
